module String exposing
    ( String
    , append, charAt, substring, length
    )

import Basics exposing (..)
import Option exposing (Option)


type String = InternalString

append : String -> String -> String
append = append

charAt : Int -> String -> Option Char
charAt = charAt

substring : Int -> Int -> String -> String
substring = substring

length : String -> Int
length = length

equals : String -> String -> Bool
equals = equals

less : String -> String -> Bool
less = less

impl String : Eq where
    (==) = equals

impl String : Ord where
    compare a b =
        if less a b then
            LT
        else if equals a b then
            EQ
        else
            GT

impl String : ToString where
    toString = id

impl String : Appendable where
    (++) = append

impl String : Default where
   default = ""
